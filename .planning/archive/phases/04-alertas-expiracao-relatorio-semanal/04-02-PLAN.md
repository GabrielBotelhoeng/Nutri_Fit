**backend/src/services/alertas.ts** — Criar com a funcao `dispararAlertas(horario: string)`:

```typescript
import { createClient } from '@supabase/supabase-js';
import { env } from '../config/env';
import { sendText } from './evolution';

const supabase = createClient(env.SUPABASE_URL, env.SUPABASE_SERVICE_KEY);

// Mensagens por tipo de alerta
const MENSAGENS: Record<string, string> = {
  cafe:        '☀️ Hora do café! Lembre-se de registrar sua refeição da manhã no NutriChat. 🥗',
  almoco:      '🌞 Hora do almoço! Não esqueça de registrar o que você comer. 🍽️',
  jantar:      '🌙 Hora do jantar! Registre sua refeição noturna no NutriChat. 🥗',
  agua:        '💧 Hora de hidratar! Beba um copo d\'água agora e registre: "bebi 300ml". 💚',
  suplemento:  '💊 Hora dos suplementos! Não esqueça de tomar e registrar no NutriChat.',
};

export async function dispararAlertas(horario: string): Promise<void> {
  console.log(`[alertas] Verificando alertas para horario ${horario}`);

  // Buscar todos os pacientes com alertas_config ativo e join com pacientes ativos
  const { data: configs, error } = await supabase
    .from('alertas_config')
    .select(`
      paciente_id,
      horario_cafe,
      horario_almoco,
      horario_jantar,
      horarios_agua,
      horario_suplementos,
      pacientes!inner(whatsapp, nome, ativo)
    `)
    .eq('ativo', true)
    .eq('pacientes.ativo', true);

  if (error) {
    console.error('[alertas] Erro ao buscar configuracoes:', error.message);
    return;
  }

  if (!configs || configs.length === 0) {
    console.log('[alertas] Nenhuma configuracao ativa — skip');
    return;
  }

  let enviados = 0;

  for (const config of configs) {
    const paciente = config.pacientes as { whatsapp: string; nome: string; ativo: boolean };
    const wpp = paciente.whatsapp;

    // Verificar cada tipo de alerta
    const alertasParaEnviar: string[] = [];

    if (config.horario_cafe === horario) alertasParaEnviar.push('cafe');
    if (config.horario_almoco === horario) alertasParaEnviar.push('almoco');
    if (config.horario_jantar === horario) alertasParaEnviar.push('jantar');

    const aguaArr = (config.horarios_agua as string[]) || [];
    if (aguaArr.includes(horario)) alertasParaEnviar.push('agua');

    const supArr = (config.horario_suplementos as string[]) || [];
    if (supArr.includes(horario)) alertasParaEnviar.push('suplemento');

    for (const tipo of alertasParaEnviar) {
      await sendText(wpp, MENSAGENS[tipo] || `Lembrete: ${tipo}`);
      console.log(`[alertas] ${tipo} enviado para ${paciente.nome} (${wpp})`);
      enviados++;
    }
  }

  console.log(`[alertas] Concluido — ${enviados} alertas enviados para horario ${horario}`);
}
```

**backend/src/routes/alertas.ts** — Criar com validacao de horario (D-05 ASVS V5):

```typescript
import { Router, Request, Response, NextFunction } from 'express';
import { env } from '../config/env';
import { dispararAlertas } from '../services/alertas';

export const alertasRouter = Router();

function requireInternalKey(req: Request, res: Response, next: NextFunction): void {
  if (req.headers['x-internal-key'] !== env.INTERNAL_API_KEY) {
    res.status(401).json({ error: 'Unauthorized' });
    return;
  }
  next();
}

// Validar formato HH:MM (00:00 a 23:59)
const HORARIO_RE = /^([01]\d|2[0-3]):[0-5]\d$/;

alertasRouter.post('/disparar', requireInternalKey, async (req: Request, res: Response) => {
  const { horario } = req.body as { horario?: string };

  if (!horario || !HORARIO_RE.test(horario)) {
    res.status(400).json({ error: 'horario invalido — formato esperado: HH:MM (ex: 07:30)' });
    return;
  }

  res.status(202).json({ status: 'processando', horario });

  dispararAlertas(horario).catch((err) => {
    console.error('[alertas] Erro ao disparar alertas:', err);
  });
});
```
