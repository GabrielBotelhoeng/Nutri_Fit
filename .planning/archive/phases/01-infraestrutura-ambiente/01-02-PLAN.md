**4. Arquivo: backend/src/config/env.ts**

Valida variaveis de ambiente na inicializacao. Se faltar alguma, o servidor NAO sobe e loga o nome da variavel ausente.

```typescript
import 'dotenv/config';

interface Env {
  PORT: number;
  NODE_ENV: string;
  SUPABASE_URL: string;
  SUPABASE_ANON_KEY: string;
  SUPABASE_SERVICE_KEY: string;
  EVOLUTION_API_URL: string;
  EVOLUTION_API_KEY: string;
  N8N_WEBHOOK_URL: string;
  CLAUDE_API_KEY: string;
  GROQ_API_KEY: string;
}

function validateEnv(): Env {
  const required = [
    'SUPABASE_URL',
    'SUPABASE_ANON_KEY',
    'SUPABASE_SERVICE_KEY',
    'EVOLUTION_API_URL',
    'EVOLUTION_API_KEY',
    'N8N_WEBHOOK_URL',
    'CLAUDE_API_KEY',
    'GROQ_API_KEY',
  ] as const;

  const missing = required.filter((key) => !process.env[key]);
  if (missing.length > 0) {
    console.error('[env] Variaveis de ambiente ausentes:', missing.join(', '));
    process.exit(1);
  }

  return {
    PORT: parseInt(process.env.PORT ?? '3001', 10),
    NODE_ENV: process.env.NODE_ENV ?? 'development',
    SUPABASE_URL: process.env.SUPABASE_URL!,
    SUPABASE_ANON_KEY: process.env.SUPABASE_ANON_KEY!,
    SUPABASE_SERVICE_KEY: process.env.SUPABASE_SERVICE_KEY!,
    EVOLUTION_API_URL: process.env.EVOLUTION_API_URL!,
    EVOLUTION_API_KEY: process.env.EVOLUTION_API_KEY!,
    N8N_WEBHOOK_URL: process.env.N8N_WEBHOOK_URL!,
    CLAUDE_API_KEY: process.env.CLAUDE_API_KEY!,
    GROQ_API_KEY: process.env.GROQ_API_KEY!,
  };
}

export const env = validateEnv();
```

```typescript
import express from 'express';
import cors from 'cors';
import { env } from './config/env';
import { healthRouter } from './routes/health';

const app = express();

// Middleware
app.use(cors());
app.use(express.json({ limit: '10mb' }));

// Rotas
app.use('/health', healthRouter);

// 404 handler
app.use((_req, res) => {
  res.status(404).json({ error: 'Not found' });
});

// Iniciar servidor
const server = app.listen(env.PORT, () => {
  console.log(`[nutrichat-backend] Rodando na porta ${env.PORT} (${env.NODE_ENV})`);
});

// Graceful shutdown
process.on('SIGTERM', () => {
  console.log('[nutrichat-backend] SIGTERM recebido, encerrando...');
  server.close(() => process.exit(0));
});

export { app, server };
```

**Arquivo: docker-compose.local.yml**
```yaml
version: "3.8"

services:
  # Postgres local com pgvector — paridade com Supabase Cloud
  postgres-local:
    image: pgvector/pgvector:pg16
    container_name: nutrichat_postgres
    restart: unless-stopped
    environment:
      POSTGRES_DB: nutrichat
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: postgres
    ports:
      - "5432:5432"
    volumes:
      - nutrichat_postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - nutrichat-net

  # Evolution API — WhatsApp
  evolution-api:
    image: evoapicloud/evolution-api:latest
    container_name: nutrichat_evolution
    restart: unless-stopped
    depends_on:
      - redis
    ports:
      - "8080:8080"
    environment:
      SERVER_URL: http://localhost:8080
      DATABASE_PROVIDER: postgresql
      DATABASE_CONNECTION_URI: postgresql://postgres:postgres@postgres-local:5432/evolution_api
      CACHE_REDIS_ENABLED: "true"
      CACHE_REDIS_URI: redis://redis:6379
      AUTHENTICATION_API_KEY: ${EVOLUTION_API_KEY:-nutrichat_local_key}
      QRCODE_COLOR: "#2D5016"
    volumes:
      - evolution_instances:/evolution/instances
    networks:
      - nutrichat-net

  # Redis — cache da Evolution API
  redis:
    image: redis:7-alpine
    container_name: nutrichat_redis
    restart: unless-stopped
    command: redis-server --appendonly yes
    ports:
      - "6379:6379"
    volumes:
      - nutrichat_redis_data:/data
    networks:
      - nutrichat-net

  # N8N — orquestracao de workflows

    image: n8nio/n8n:latest
    container_name: nutrichat_n8n
    restart: unless-stopped

      - "5678:5678"

      DB_TYPE: postgresdb
      DB_POSTGRESDB_HOST: postgres-local
      DB_POSTGRESDB_PORT: 5432
      DB_POSTGRESDB_DATABASE: n8n
      DB_POSTGRESDB_USER: postgres
      DB_POSTGRESDB_PASSWORD: postgres
      N8N_BASIC_AUTH_ACTIVE: "true"
      N8N_BASIC_AUTH_USER: admin
      N8N_BASIC_AUTH_PASSWORD: ${N8N_BASIC_AUTH_PASSWORD:-nutrichat123}

      SUPABASE_ANON_KEY: ${SUPABASE_ANON_KEY}
      SUPABASE_SERVICE_KEY: ${SUPABASE_SERVICE_KEY}

      - nutrichat-net

volumes:
  nutrichat_postgres_data:
```
